class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        w=0
        r=1
        c=1
        while 1:
            if w==len(nums)-1 or r>=len(nums):
                break
            if nums[w]==nums[r]:
                r+=1
                continue
            else:
                c+=1
                w += 1
                nums[w] = nums[r]
                r+=1
                continue
            
        return c
